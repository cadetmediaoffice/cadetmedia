Step 1. Open your Google Sheets (linked to your Form responses).
Step 2. Click Extensions → Apps Script.
Step 3. Delete everything in the editor and paste this code:

function doGet(e){ var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); var data = sheet.getDataRange().getValues(); var csv = data.map(function(row){ return row.map(function(cell){ var s = String(cell); if(s.indexOf(',')>-1||s.indexOf('"')>-1||s.indexOf('\n')>-1) s = '"'+s.replace(/"/g,'""')+'"'; return s; }).join(','); }).join('\n'); return ContentService.createTextOutput(csv) .setMimeType(ContentService.MimeType.TEXT); }
